public class Example {
    
    public void extractFromPair() {
        Outcome outcome = new Outcome();
        Pair<Outcome, String> result = ImmutablePair.of(outcome, "Processing done");
        
        Outcome retrievedOutcome = result.getLeft();
        String retrievedMessage = result.getRight();
        
        System.out.println("Message: " + retrievedMessage);
        System.out.println("Outcome: " + retrievedOutcome);
    }
}
