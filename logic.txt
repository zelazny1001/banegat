def calculate_weights_difference(df1_weights, df2_weights):
  weights_difference = df1_weights.subtract(df2_weights)
  return weights_difference

def visualize_weights_difference(weights_difference):
  plt.figure(figsize=(12, 8))
  sns.heatmap(weights_difference, cmap='RdBu_r', center=0, annot=True, fmt='.2f')
  plt.title('Difference in Weights Between Model 1 and Model 2')
  plt.xlabel('Intents')
  plt.ylabel('Features')
  plt.tight_layout()
  plt.show()
